        选中的id: {{ selectId }}

        id: 232
    }
]);

const row = ref(data.value[0]);
const change = (record: any) => {
    row.value = record;
};
</script>

spinning加载进度 ​

加载中...

No data

spinning
设置spinning属性显示加载动画
<template>
    <div class="vp-raw">
        <a-button @click="load">加载</a-button>
        <w-select-card v-model="selectId" :data="data" :spinning="spinning">
            <template #default="{ record }">
                <div>{{ record.name }}</div>
                <div>{{ record.age }}</div>
            </template>
        </w-select-card>
    </div>
</template>
<script lang="ts" setup>
import { ref } from "vue";
const data = ref();
const selectId = ref(23);
const spinning = ref(false);
const load = () => {
    spinning.value = true;
    setTimeout(() => {
        data.value = [
            {
                name: "张三",
                age: 18,
                id: 23
            },
            {
                name: "张三",
                age: 18,
                id: 231
            },
            {
                name: "张三",
                age: 18,
                id: 232
            }
        ];
        spinning.value = false;
    }, 1000);
};
load();
</script>

属性 ​

参数说明类型默认值
data数据集合Array[]
v-model(modelValue)绑定选中的数据，以rowKey设置绑定的值，使用multiple则为数组String|Number|Stirng[]|Number[]
rowKey设置modelValue绑定的值Stringid
spinning加载动画booleanfalse
tip加载动画提示文字String加载中...
column列表列数Number3
bg背景，颜色或者图片都可以String
activeColor设置选中激活色String#1890ff
disabled禁用所有选中，不建议使用该属性，因为这样就丢失了该组件的主要场景，当然如果需要设置某一列禁用则在数据data中添加disabled为true即可Booleanfalse
multiple是否多选Booleanfalse
cancel是否取消选中Booleanfalse
