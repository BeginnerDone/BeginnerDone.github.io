介绍 ​

@wlink/config-provider 是一个配置组件，用于配置@wlink/ui和@wlink/antd-next以及@wlink/element-ui的组件的默认配置。

WARNING

在使用 @wlink/antd-next以及@wlink/element-ui的组件时，必须使用@wlink/config-provider 来配置select、tree-select、table等该工程的组件的默认配置。

示例 ​

fetchSelect.ts：

ts
import request from "../../mock/http/request";
import type { MethodEnum } from "wlink-ui/config-provider/src/enum";
// 下拉数据
export function fetchSelect(
    url: string,
    method: MethodEnum,
    params: any,
    isBody = false
) {
    if (method === "get") {
        return request.get(url, { params: params });
    } else if (method === "post") {
        if (isBody) {
            return request.request({
                url: url,
                method: "post",
                data: params
            });
        } else {
            return request.post(url, params);
        }
    } else {
        return request.post(url);
    }
}

Config-Provider.ts 配置：

ts
import { ConfigProvider } from "@wlink/config-provider";
import { fetchSelect } from "./api/select";
import { fetchTreeSelect } from "./api/tree-select";
import { fetchTable } from "./api/table";
import { fetchTree } from "./api/tree";

export function useConfig() {
    ConfigProvider.config({
        vxeConfig: {
            grid: {
                exportConfig: {
                    type: "xlsx"
                },
                checkboxConfig: {
                    highlight: true,
                    range: true
                },
                sortConfig: {
                    // 配合columns中的sortable和proxy-config中的sort来设置服务器排序
                    remote: true,
                    trigger: "cell"
                },
                toolbarConfig: {
                    custom: true,
                    print: true,
                    export: true,
                    refresh: true,
                    zoom: true
                }
            }
        },
        antd: {
            dialog: {
                showMini: false,
                showMax: false,
                miniStyle: {}
            },
            select: {
                request: (
                    url: string,
                    method: string,
                    params: any,
                    isBody: boolean
                ) => {
                    return new Promise((resolve, reject) => {
                        fetchSelect(url, method, params, isBody).then(res => {
                            resolve(res.data);
                        });
                    });
                },
                // 配置select公共的字段
                fieldNames: {
                    label: "name",
                    value: "value"
                }
            },
            treeSelect: {
                request: (
                    url: string,
                    method: string,
                    params: any,
                    isBody: boolean
                ) => {
                    return new Promise((resolve, reject) => {
                        fetchTreeSelect(url, method, params, isBody).then(
                            res => {
                                resolve(res.data);
                            }
                        );
                    });
                },
                fieldNames: {
                    label: "name",
                    value: "value",
                    children: "children"
                }
            },
            table: {
                request: (
                    url: string,
                    method: string,
                    page: number | undefined,
                    size: number | undefined,
                    params: any,
                    isBody: boolean
                ) => {
                    return new Promise((resolve, reject) => {
                        fetchTable(
                            url,
                            method,
                            page,
                            size,
                            params,
                            isBody
                        ).then((res: any) => {
                            resolve({ dataSource: res.data, total: res.total });
                        });
                    });
                }
            },
            tree: {
                request: (
                    url: string,
                    method: string,
                    params: any,
                    isBody: boolean
                ) => {
                    return new Promise((resolve, reject) => {
                        fetchTree(url, method, params, isBody).then(res => {
                            resolve(res.data);
                        });
                    });
                },
                fieldNames: {
                    title: "name",
                    key: "value",
                    children: "children"
                }
            }
        }
    });
}

TableConfig

ts
import type { MethodEnum } from "../enum";

type method = keyof typeof MethodEnum;

type SizeType = "small" | "large" | "middle" | undefined;

interface ResponseData {
    // res返回的集合数据
    dataSource: any[];
    // res返回的总数
    total: number;
}

/**
 * @param url 请求地址
 * @param method 请求方法
 * @param params 请求参数
 * @param isBody 是否是body请求
 * @param page 分页参数
 * @param size 分页大小
 */
export interface TableRequestFunction {
    (
        url: string,
        method: method,
        page: number | undefined,
        size: number | undefined,
        params: any,
        isBody: boolean
    ): Promise<ResponseData>;
}

/**
 * @param showReload 是否显示刷新按钮
 * @param showSize 是否显示表格尺寸选择器
 */
export interface TableTitleConfig {
    showReload: boolean;
    showSize: boolean;
}
export interface TableConfig {
    request: TableRequestFunction;
    // 表格标题
    titleConfig?: TableTitleConfig;
    // 表格尺寸
    size?: SizeType;
    // attrs，配置表格属性。例如：{border: true}，其他公共属性不建议放置于次
    attrs?: Record<string, any>;
}

        grid: {
            // 表格尺寸

            // 工具栏导出的配置
            exportConfig: {

                highlight: true
            },
            toolbarConfig: {
                custom: true,
                print: true,
                export: true,
                // refresh: true,
                zoom: true,
                // 表格自定义按钮
                slots: {
                    // 工具栏左侧插槽
                    buttons: "toolbar_buttons",
                    // 工具栏右侧插槽(这个不行，会覆盖上面的print zoom等。)
                    tools: "toolbar_tools"
                }
            }
